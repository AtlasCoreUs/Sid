class AdaptiveAbbreviationAI {
  // Base de données évolutive
  userAbbreviations = new Map<string, AbbreviationPattern>();
  globalPatterns = new CommunityLearning();
  
  // Apprentissage en temps réel
  learnFromUser(userInput: string, correctedOutput: string, userId: string): void {
    const newPatterns = this.extractAbbreviationPatterns(userInput, correctedOutput);
    
    newPatterns.forEach(pattern => {
      // Enregistre pour cet utilisateur
      this.userAbbreviations.set(pattern.abbreviation, {
        fullForm: pattern.fullForm,
        confidence: 1.0,
        userId: userId,
        context: pattern.context,
        frequency: 1,
        dateAdded: new Date()
      });
      
      // Contribue à l'intelligence collective
      this.globalPatterns.addPattern(pattern, userId);
    });
  }
  
  // Détection automatique nouvelles abréviations
  detectNewAbbreviations(handwriting: string): NewAbbreviation[] {
    return this.analyzeUnknownPatterns(handwriting)
      .filter(pattern => this.isLikelyAbbreviation(pattern))
      .map(pattern => ({
        detected: pattern.text,
        suggestions: this.generateExpansionSuggestions(pattern),
        confidence: this.calculateConfidence(pattern)
      }));
  }
}

class PersonalAbbreviationPredictor {
  // Prédit les abréviations que l'user va créer
  predictNextAbbreviations(userHistory: UserPattern): Prediction[] {
    const patterns = this.analyzeUserTendencies(userHistory);
    
    return [
      // Basé sur longueur mots préférée
      ...this.predictByWordLength(patterns.preferredWordLength),
      
      // Basé sur fréquence d'usage
      ...this.predictByFrequency(patterns.mostUsedWords),
      
      // Basé sur domaine d'étude
      ...this.predictByDomain(patterns.studyDomain),
      
      // Basé sur style personnel
      ...this.predictByStyle(patterns.abbreviationStyle)
    ];
  }
  
  // Suggestions proactives
  suggestAbbreviations(currentText: string, userContext: Context): void {
    const longWords = this.findLongRepetitiveWords(currentText);
    
    longWords.forEach(word => {
      if (this.shouldSuggestAbbreviation(word, userContext)) {
        this.showAbbreviationSuggestion({
          word: word,
          suggestions: this.generateAbbreviations(word),
          reason: `Tu écris "${word}" souvent, veux-tu créer une abréviation ?`
        });
      }
    });
  }
}
